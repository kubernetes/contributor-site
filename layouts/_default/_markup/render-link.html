{{- $u := .Destination -}} {{- /* Handle extra whitespace or parentheses sometimes found in external content */ -}} {{- $u = trim $u " ()" -}} {{- $isExternal := strings.HasPrefix $u "http" -}} {{- $isFragment := strings.HasPrefix $u "#" -}} {{- $isMailto := strings.HasPrefix $u "mailto:" -}} {{- if and (not $isExternal) (not $isFragment) (not $isMailto) -}} {{- /* Try Hugo's native page resolution */ -}} {{- $page := .Page.GetPage $u -}} {{- if $page -}} {{- $u = $page.RelPermalink -}} {{- else -}} {{- /* Fallback to external repository for non-local links */ -}} {{- if not (strings.HasPrefix $u "/") -}} {{- /* Normalize relative paths (resolve ../ traversal) before building URL */ -}} {{- $normalized := $u -}} {{- $parts := split $u "/" -}} {{- $resolved := slice -}} {{- range $parts -}} {{- if eq . ".." -}} {{- if gt (len $resolved) 0 -}} {{- $resolved = $resolved | first (sub (len $resolved) 1) -}} {{- end -}} {{- else if and (ne . "") (ne . ".") -}} {{- $resolved = $resolved | append . -}} {{- end -}} {{- end -}} {{- if gt (len $resolved) 0 -}} {{- $normalized = delimit $resolved "/" -}} {{- end -}} {{- /* NOTE: Upstream repos currently use 'master'. Update if renamed to 'main'. */ -}} {{- if strings.Contains .Page.File.Dir "docs/guide" -}} {{- $u = printf "https://github.com/kubernetes/community/blob/master/contributors/guide/%s" $normalized -}} {{- else if strings.Contains .Page.File.Dir "docs/comms" -}} {{- $u = printf "https://github.com/kubernetes/community/blob/master/communication/%s" $normalized -}} {{- else if strings.Contains .Page.File.Dir "docs/orientation" -}} {{- $u = printf "https://github.com/kubernetes/community/blob/master/mentoring/new-contributor-orientation/%s" $normalized -}} {{- else if strings.Contains .Page.File.Dir "community" -}} {{- $u = printf "https://github.com/kubernetes/community/blob/master/%s" $normalized -}} {{- else if strings.Contains .Page.File.Dir "resources/release" -}} {{- $u = printf "https://github.com/kubernetes/sig-release/blob/master/releases/%s" $normalized -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- .Text | safeHTML -}}
